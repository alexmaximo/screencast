(function() {
  angular.module("screencast").controller("UserController", function($scope, githubContributor) {
    $scope.users = [];
    return $scope.getContributors = function(limit) {
      var errorFn, successFn;
      successFn = function(result) {
        return $scope.users = result.data;
      };
      errorFn = function(error) {
        return console.log(error.error);
      };
      if (!limit) {
        limit = 10;
      }
      return githubContributor.getContributors(limit).then(successFn, errorFn);
    };
  });
 
}).call(this);
