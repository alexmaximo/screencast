(function() {
  angular.module("screencast").factory('githubContributor', function($log, $http) {
    var apiHost, getContributors, service;
    apiHost = 'https://api.github.com/repos/Swiip/generator-gulp-angular';
    getContributors = function(limit) {
      return $http.get(apiHost + '/contributors?per_page=' + limit);
    };
    return service = {
      apiHost: apiHost,
      getContributors: getContributors
    };
  });
 
}).call(this);
